algoritmo "SOMA-SUB-DIV-FUNÇAO"

var
    V1,V2,S,SD,D:INTEIRO


FUNCAO SOMA(X,Y:INTEIRO):INTEIRO
INICIO

   RETORNE X+Y
      
FIMFUNCAO
FUNCAO SOMAD(X1,Y1:INTEIRO):INTEIRO
INICIO

   RETORNE X1-Y1
      
FIMFUNCAO
FUNCAO DIVISAO(X2,Y2:INTEIRO):INTEIRO
INICIO

   RETORNE X2 DIV Y2

FIMFUNCAO
inicio
   ESCREVA("DIGITE UM NUMERO: ")
   LEIA(V1)
   ESCREVA("DIGITE OUTRO NUMERO: ")
   LEIA(V2)
   S<-SOMA(V1,V2)
   SD<-SOMAD(V1,V2)
   D<-DIVISAO(V1,V2)
   ESCREVAL("A SOMA ENTRE",V1," E ",V2," É IGUAL A ",S)
   ESCREVAL("A SUBTRAÇAO ENTRE OS VALORES É",SD)
   ESCREVAL("A DIVISAO DE ",V1," E ",V2," É ", D)
