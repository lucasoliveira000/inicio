      escreval("LUGARES NA SALA")
      escreval("")
      
        para cont1<-1 ate 10 faca
          se(assento[cont1]=0)entao
            escreva("")
            escreva(" ","[B",cont1,"]")
          senao
            escreva("[----]")
          fimse
        fimpara

      escreval("")
      escreval("")

      escreva("escolha um assento B:")

        se(assento[escolhido]=0)entao
           assento[escolhido]<-1
           escreval("lugar reservado!")

           escreval("erro!lugar ocupado")
        fimse

      leia(continuar)
