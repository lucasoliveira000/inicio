algoritmo "SEGUIMENTACAO-SITE"

var
   IDADE,CLIENTE1,CLIENTE2,CLIENTE3,CONT1,CONT2,CONT3:INTEIRO
   TRABALHO,LAZER,FR:CARACTERE

   
FUNCAO FAIXAETARIA(IDADE:INTEIRO):INTEIRO
VAR
   CLIENTE1:INTEIRO
INICIO
   SE(IDADE>=18)E(IDADE<=30)ENTAO
      CLIENTE1<-1
   SENAO
      CLIENTE1<-2
   FIMSE
    RETORNE CLIENTE1
FIMFUNCAO

FUNCAO TRABALHANDO(TRABALHO:CARACTERE):INTEIRO
VAR
   CLIENTE2:INTEIRO
INICIO
   SE(TRABALHO="SIM")ENTAO
      CLIENTE2<-1
   SENAO
      CLIENTE2<-2
   FIMSE
   RETORNE CLIENTE2
FIMFUNCAO

FUNCAO OPCAOLAZER(LAZER:CARACTERE):INTEIRO
VAR
   CLIENTE3:INTEIRO
INICIO
      SE(LAZER="CASA")ENTAO
         CLIENTE3<-1
      SENAO
         CLIENTE3<-2
      FIMSE
      RETORNE CLIENTE3
FIMFUNCAO
PROCEDIMENTO CONTADOR()
INICIO
         SE(CLIENTE1=1)E (CLIENTE2=1)E(CLIENTE3=2)ENTAO
            CONT1<-CONT1+1
         FIMSE
         
         SE(CLIENTE2=2)E(CLIENTE1=2)E(CLIENTE3=2)ENTAO
            CONT2<-CONT2+1
         FIMSE

         SE(CLIENTE3=1)E(CLIENTE2=1)E(CLIENTE1=1)ENTAO
            CONT3<-CONT3+1
         FIMSE
FIMPROCEDIMENTO

inicio
   CLIENTE1<-0
   CLIENTE2<-0
   CLIENTE3<-0
   REPITA
         ESCREVAL("=======================")
         ESCREVAL("BEM VINDO AO -OPINIONS-")
         ESCREVAL("=======================")
   
         ESCREVA("QUAL SUA IDADE?: ")
         LEIA(IDADE)
         ESCREVAL("VOCÊ TRABALHA?[SIM]/[NÃO]: ")
         LEIA(TRABALHO)
         ESCREVA("ENTRE ESTAR EM CASA E SAIR QUAL PREFERE?[CASA]/[SAIR]: ")
         LEIA(LAZER)
         CLIENTE1<-FAIXAETARIA(IDADE)
         CLIENTE2<-TRABALHANDO(TRABALHO)
         CLIENTE3<-OPCAOLAZER(LAZER)
         CONTADOR()
         LIMPATELA
         ESCREVA("CONTINUAR COM A PESQUISA?[SIM]/[NAO]: ")
         LEIA(FR)
   ATE(FR="NAO")
         LIMPATELA
         ESCREVAL("O NUMEROD DE PESSOAS QUE TEM IDADE +18 ,TRABALHAM E PREFEREM SAIR É:",CONT1)
         ESCREVAL("")
         ESCREVAL("O NUMERO DE PESSOAS QUE TEM IDADE  -18 ,NAO TRABALHAM E PREFEREM SAIR É:",CONT2)
         ESCREVAL("")
         ESCREVAL("O NUMERO DE PESSOAS QUE TEM IDADE  +18 ,TRABALHAM E PREFEREM FICAR EM CASA É:",CONT3)
